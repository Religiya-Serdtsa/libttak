1/**
2 * @file security_engine.h
3 * @brief Unified cryptographic dispatch engine for LibTTAK.
4 *
5 * Provides a single entry point (@c ttak_security_execute) that routes
6 * encryption, hashing, and KDF requests to the best available driver
7 * (scalar, SIMD, or hardware accelerator) selected at runtime.
8 *
9 * Supported operations: LEA, SEED, AES-256-GCM, ChaCha20-Poly1305,
10 * post-quantum signatures, fast hashing, and memory-hard KDF.
11 */

57/**
58 * @brief Unified I/O context passed to every cipher and hash operation.
59 *
60 * Callers populate the relevant fields for their operation and pass this
61 * structure to ttak_security_execute(). Fields that do not apply to a
62 * given operation may be left zeroed.
63 */
64typedef struct ttak_crypto_ctx {
65 const uint8_t *in; /**< Plaintext / input buffer. */
66 size_t in_len; /**< Length of @c in in bytes. */
67 uint8_t *out; /**< Ciphertext / output buffer. */
68 size_t out_len; /**< Capacity of @c out in bytes. */
69 const uint8_t *key; /**< Raw key material. */
70 size_t key_len; /**< Key length in bytes. */
71 void *scratch; /**< Optional caller-provided scratch space. */
72 size_t scratch_len; /**< Size of @c scratch in bytes. */
73 const uint8_t *aad; /**< Additional authenticated data (AEAD). */
74 size_t aad_len; /**< Length of @c aad in bytes. */
75 uint8_t iv[16]; /**< Initialisation vector / nonce. */
76 size_t iv_len; /**< Active IV length in bytes. */
77 uint8_t *tag; /**< Authentication tag output (AEAD). */
78 size_t tag_len; /**< Desired tag length in bytes. */
79 const uint8_t **in_blocks; /**< Optional scatter-gather input array. */
80 uint8_t **out_blocks; /**< Optional scatter-gather output array. */
81 size_t block_count; /**< Number of scatter-gather blocks. */
82 size_t block_size; /**< Size of each scatter-gather block. */
83 union {
84 struct {
85 alignas(16) uint8_t round_keys[15][16]; /**< AES-256 round keys (15 × 16 B). */
86 size_t rounds; /**< Active round count (14 for AES-256). */
87 } aes;
88 struct {
89 uint32_t round_keys[32]; /**< SEED expanded round keys. */
90 size_t rounds; /**< Active round count (16 for SEED-128). */
91 } seed;
92 } hw_state; /**< Pre-expanded cipher state for hardware-assisted paths. */
93} ttak_crypto_ctx_t;
