19/**
20 * @brief Bitmask flags representing the current status of the shared resource.
21 */
22typedef uint32_t ttak_shared_status_t;
23#define TTAK_SHARED_READY (0) /**< Resource is stable and ready for use */
24#define TTAK_SHARED_DIRTY (1 << 0) /**< Data has been modified but not synchronized */
25#define TTAK_SHARED_EXPIRED (1 << 1) /**< Resource timestamp is outdated */
26#define TTAK_SHARED_ZOMBIE (1 << 2) /**< No active owners; pending deallocation */
27#define TTAK_SHARED_READONLY (1 << 3) /**< Resource is in read-only mode */
28#define TTAK_SHARED_USE_EBR (1 << 4) /**< EBR protection enabled */
29#define TTAK_SHARED_SWAPPING (1 << 5) /**< Data is currently being swapped */

31/**
32 * @brief Operational result codes for shared memory actions.
33 */
34typedef uint32_t ttak_shared_result_t;
35#define TTAK_OWNER_VALID (0) /**< Level 3: Pure safety guaranteed */
36#define TTAK_OWNER_CORRUPTED (1 << 0) /**< Level 2: Timestamp mismatch detected */
37#define TTAK_OWNER_INVALID (1 << 1) /**< Level 1: Owner information mismatch */
38#define TTAK_OWNER_SHARE_DENIED (1 << 2) /**< Error: Failed to register new owner */
39#define TTAK_OWNER_SUCCESS (1 << 3) /**< Operation completed successfully */

41/**
42 * @brief Security enforcement levels for ownership validation.

44typedef enum {
45 TTAK_SHARED_LEVEL_3, /**< Strict: Full ownership and timestamp validation */
46 TTAK_SHARED_LEVEL_2, /**< Moderate: Allows slight timestamp drift */
47 TTAK_SHARED_LEVEL_1, /**< Loose: Basic owner check only; not recommended */
48 TTAK_SHARED_NO_LEVEL /**< None: No ownership check; maximum performance/danger */
