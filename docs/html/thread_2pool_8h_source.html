1 /**
2  * @file pool.h
3  * @brief Managed thread pool built on top of the async task scheduler.
4  *
5  * Provides a fixed-size pool of worker threads that drain a sharded task
6  * queue. Tasks are routed to shards via a deterministic hash-to-coordinate
7  * mapping backed by a Latin-square routing table, reducing lock contention
8  * by confining queue operations to a single shard lock rather than a global
9  * pool lock. Threads are created lazily and respect the nice value set at
10  * init time.
11  */
