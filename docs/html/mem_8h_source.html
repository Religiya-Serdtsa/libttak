63/**
64 * @struct ttak_mem_header_t

67 * 64-byte aligned to prevent False Sharing and ensure user pointer alignment.

187/**
188 * @brief Accesses a memory block, verifying its lifecycle and security.
189 * @param ptr Pointer to user memory.
190 * @param now_tick Current timestamp in ticks.
191 * @return Validated pointer, or NULL if security check fails or block is expired.
192 */
193static inline void *ttak_mem_access(void *ptr, uint64_t now_tick) {
194 if (!ptr) return NULL;
195 ttak_mem_header_t *header = (ttak_mem_header_t *)ptr - 1;
196
197 if (header->magic != TTAK_MAGIC_NUMBER) return NULL;
198 if (header->freed) return NULL;
199 if (header->expires_tick != __TTAK_UNSAFE_MEM_FOREVER__ && now_tick > header->expires_tick) return NULL;
200 if (!header->allow_direct_access) return NULL;
201
202 TTAK_ATOMIC_FETCH_ADD_U64(&header->access_count, 1ULL);
203 return ptr;
204}

248/**
249 * @brief Calculates a 32-bit checksum for the memory header.
250 * @param h Pointer to the header.
251 * @return Calculated checksum.
252 */
253static inline uint32_t ttak_calc_header_checksum(const ttak_mem_header_t *h) {
254 uint32_t sum1 = h->magic;
255 uint32_t sum2 = (uint32_t)h->created_tick;
256 sum1 ^= (uint32_t)(h->created_tick >> 32);
257 sum2 ^= (uint32_t)h->expires_tick;
258 sum1 ^= (uint32_t)(h->expires_tick >> 32);
259 sum2 ^= (uint32_t)h->size;
260#if defined(__LP64__) || defined(_WIN64)
261 sum1 ^= (uint32_t)(h->size >> 32);
262#endif
263 sum2 ^= (uint32_t)h->should_join;
264 sum1 ^= (uint32_t)h->strict_check;
265 sum2 ^= (uint32_t)h->is_root;
266 sum1 ^= (uint32_t)h->canary_start;
267 sum2 ^= (uint32_t)(h->canary_start >> 32);
268 sum1 ^= (uint32_t)h->canary_end;
269 sum2 ^= (uint32_t)(h->canary_end >> 32);
270 sum1 ^= (uint32_t)h->allocation_tier;
271 return sum1 ^ sum2;
272}
