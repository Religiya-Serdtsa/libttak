93static inline void *ttak_mem_access(void *ptr, uint64_t now) {
94 if (!ptr) return NULL;
95 ttak_mem_header_t *header = (ttak_mem_header_t *)ptr - 1;
96
97 // Fast path: Optimistic lock-free check
98 if (header->magic != TTAK_MAGIC_NUMBER) return NULL; // Basic sanity
99 if (header->freed) return NULL;
100 if (header->expires_tick != __TTAK_UNSAFE_MEM_FOREVER__ && now > header->expires_tick) return NULL;
101 if (!header->allow_direct_access) return NULL;
102
103 // Atomic increment without lock
104 // Using GCC/Clang builtin for speed if available, else ttak wrapper
105 // ttak_atomic_inc64 is likely a wrapper.
106 // Let's use the library's function.
107 // Since it's inline, we need to ensure ttak_atomic_inc64 is visible. It is (included above).
108 ttak_atomic_inc64(&header->access_count);
109
110 return ptr;
111}

143/**
144 * @brief Calculates a 32-bit checksum for the memory header.
145 */
146static inline uint32_t ttak_calc_header_checksum(const ttak_mem_header_t *h) {
147 uint32_t sum1 = h->magic;
148 uint32_t sum2 = (uint32_t)h->created_tick;
149 sum1 ^= (uint32_t)(h->created_tick >> 32);
150 sum2 ^= (uint32_t)h->expires_tick;
151 sum1 ^= (uint32_t)(h->expires_tick >> 32);
152 sum2 ^= (uint32_t)h->size;
153#if defined(__LP64__) || defined(_WIN64)
154 sum1 ^= (uint32_t)(h->size >> 32);
155#endif
156 sum2 ^= (uint32_t)h->should_join;
157 sum1 ^= (uint32_t)h->strict_check;
158 sum2 ^= (uint32_t)h->is_root;
159 sum1 ^= (uint32_t)h->canary_start;
160 sum2 ^= (uint32_t)(h->canary_start >> 32);
161 sum1 ^= (uint32_t)h->canary_end;
162 sum2 ^= (uint32_t)(h->canary_end >> 32);
163 return sum1 ^ sum2;
164}
