70/**
71 * @struct ttak_mem_header_t
72 * @brief "Fortress" Memory Header stored before user data.
73 *
74 * 64-byte aligned to prevent False Sharing and ensure user pointer alignment.
75 * The structure is padded to maintain alignment for the following user data.
76 */
77typedef struct ttak_mem_header_t {
78 alignas(64) uint32_t magic; /**< 0x5454414B */
79 uint32_t checksum; /**< Metadata checksum to detect header corruption */
80 uint64_t created_tick; /**< Creation timestamp in ticks */
81 uint64_t expires_tick; /**< Expiration timestamp in ticks */
82 uint64_t access_count; /**< Atomic access audit counter */
83 uint64_t pin_count; /**< Atomic reference count for pinning */
84 size_t size; /**< User-requested size in bytes */
85 pthread_mutex_t lock; /**< Per-header synchronization lock */
86 uint8_t freed; /**< True if the block has been deallocated */
87 uint8_t is_const; /**< Immutability hint */
88 uint8_t is_volatile; /**< Volatility hint */
89 uint8_t allow_direct_access; /**< Safety bypass flag for direct pointer access */
90 uint8_t is_huge; /**< True if mapped via hugepages */
91 uint8_t should_join; /**< Indicates if associated resource needs joining */
92 uint8_t strict_check; /**< Enable strict memory boundary (canary) checks */
93 uint8_t is_root; /**< Marks the allocation as a root node for the mem_tree */
94 uint64_t canary_start; /**< Magic number for start of user data (in strict mode) */
95 uint64_t canary_end; /**< Magic number for end of user data (in strict mode) */

97 uint8_t allocation_tier; /**< Tier that performed the allocation */
98 size_t mapped_size; /**< Total OS-mapped bytes (header + payload + canary) */
99 char reserved[2]; /**< Explicit padding for header alignment */
100} ttak_mem_header_t;

208/**
209 * @brief Accesses a memory block, verifying its lifecycle and security.
210 * @param ptr Pointer to user memory.
211 * @param now_tick Current timestamp in ticks.
212 * @return Validated pointer, or NULL if security check fails or block is expired.
213 */
214#if defined(TTAK_MEM_FORCE_ACCESS_BRIDGE)
215static inline void *ttak_mem_access(void *ptr, uint64_t now_tick) {
216 return ttak_mem_access_bridge(ptr, now_tick);
217}
218#else
219static inline void *ttak_mem_access(void *ptr, uint64_t now_tick) {
220 if (!ptr) return NULL;
221 ttak_mem_header_t *header = (ttak_mem_header_t *)ptr - 1;
222
223 if (header->magic != TTAK_MAGIC_NUMBER) return NULL;
224 if (header->freed) return NULL;
225 if (header->expires_tick != __TTAK_UNSAFE_MEM_FOREVER__ && now_tick > header->expires_tick) return NULL;
226 if (!header->allow_direct_access) return NULL;
227
228 TTAK_ATOMIC_FETCH_ADD_U64(&header->access_count, 1ULL);
229 return ptr;
230}
231#endif

275/**
276 * @brief Calculates a 32-bit checksum for the memory header.
277 * @param h Pointer to the header.
278 * @return Calculated checksum.
279 */
280static inline uint32_t ttak_calc_header_checksum(const ttak_mem_header_t *h) {
281 uint32_t sum1 = h->magic;
282 uint32_t sum2 = (uint32_t)h->created_tick;
283 sum1 ^= (uint32_t)(h->created_tick >> 32);
284 sum2 ^= (uint32_t)h->expires_tick;
285 sum1 ^= (uint32_t)(h->expires_tick >> 32);
286 sum2 ^= (uint32_t)h->size;
287#if defined(__LP64__) || defined(_WIN64)
288 sum1 ^= (uint32_t)(h->size >> 32);
289#endif
290 sum2 ^= (uint32_t)h->should_join;
291 sum1 ^= (uint32_t)h->strict_check;
292 sum2 ^= (uint32_t)h->is_root;
293 sum1 ^= (uint32_t)h->canary_start;
294 sum2 ^= (uint32_t)(h->canary_start >> 32);
295 sum1 ^= (uint32_t)h->canary_end;
296 sum2 ^= (uint32_t)(h->canary_end >> 32);
297 sum1 ^= (uint32_t)h->allocation_tier;
298 return sum1 ^ sum2;
299}
