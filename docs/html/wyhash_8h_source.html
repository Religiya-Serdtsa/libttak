15static inline uint64_t _ttak_wymum(uint64_t A, uint64_t B) {
16#ifdef __SIZEOF_INT128__
17 __uint128_t r = (__uint128_t)A * B;
18 return (uint64_t)r ^ (uint64_t)(r >> 64);
19#else
20 uint64_t ha = A >> 32, hb = B >> 32, la = (uint32_t)A, lb = (uint32_t)B;
21 uint64_t rh = ha * hb, rm0 = ha * lb, rm1 = hb * la, rl = la * lb;
22 uint64_t t = rl + (rm0 << 32), c = t < rl;
23 uint64_t lo = t + (rm1 << 32), hi = rh + (rm0 >> 32) + (rm1 >> 32) + c + (lo < t);
24 return hi ^ lo;
25#endif
26}
