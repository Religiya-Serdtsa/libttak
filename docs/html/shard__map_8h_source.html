2  * @file shard_map.h
3  * @brief Deterministic shard mapping utilities for the concurrent async runtime.
4  *
5  * Implements a deterministic, array-based computation model for routing task
6  * hashes to shard indices. The mapping pipeline is:
7  *
8  * task_hash → (row, col) coordinates → routing table lookup → shard index
9  *
10  * The routing table is pre-computed as an 8×8 Latin square, guaranteeing that
11  * every shard appears exactly once in each row and each column. This provides
12  * uniform dispersion similar in spirit to Orthogonal Latin Square constructions,
13  * minimising clustering and collision across shards.
14  *
15  * All functions are deterministic, pure, and side-effect-free.
16  */

33 /**
34  * @brief 8×8 Latin-square routing table.
35  *
36  * shard_route_table[row][col] gives the shard index for a task whose
37  * hash coordinates are (row, col). Each value 0-7 appears exactly once
38  * per row and once per column, ensuring uniform distribution and avoiding
39  * hot-shard clustering for structured hash inputs.
40  *
41  * Construction: entry (r, c) = (r + c) % 8 (cyclic Latin square).

55 /**
56  * @brief Convert a task hash into (row, col) shard coordinates.
57  *
58  * Uses Fibonacci/multiplicative hashing with the 64-bit golden-ratio constant
59  * to project the hash into a bounded 2D coordinate space. The upper and lower
60  * halves of the mixed value are used independently to decorrelate the two axes.
61  *
62  * @param hash Task hash value.
63  * @param row Output: row coordinate in [0, TTAK_POOL_SHARD_COUNT).
64  * @param col Output: column coordinate in [0, TTAK_POOL_SHARD_COUNT).
65  */
66 static inline void ttak_shard_hash_to_coords(uint64_t hash,
67  uint32_t *row,
68  uint32_t *col)
69 {
70  /* 64-bit golden-ratio constant (⌊2^64 / φ⌋) for Fibonacci hashing */
71  const uint64_t GOLDEN = UINT64_C(0x9e3779b97f4a7c15);
72  uint64_t mixed = hash * GOLDEN;
73  /* Upper 32 bits → row; lower 32 bits → col.
74  * Both are masked to [0, TTAK_POOL_SHARD_COUNT). */
75  *row = (uint32_t)(mixed >> 32) & (uint32_t)TTAK_POOL_SHARD_MASK;
76  *col = (uint32_t)(mixed & UINT32_MAX) & (uint32_t)TTAK_POOL_SHARD_MASK;
77 }
