1/**
2 * @file shard_map.h
3 * @brief Deterministic shard mapping utilities for the concurrent async runtime.
4 *
5 * Implements a deterministic, array-based computation model for routing task
6 * hashes to shard indices. The mapping pipeline is:
7 *
8 * task_hash → (row, col) coordinates → routing table lookup → shard index
9 *
10 * The routing table is pre-computed as an 8×8 Latin square, guaranteeing that
11 * every shard appears exactly once in each row and each column. This provides
12 * uniform dispersion similar in spirit to Orthogonal Latin Square constructions,
13 * minimising clustering and collision across shards.
14 *
15 * All functions are deterministic, pure, and side-effect-free.
16 */
17

34/**
35 * @brief 8×8 Latin-square routing table.
36 *
37 * shard_route_table[row][col] gives the shard index for a task whose
38 * hash coordinates are (row, col). Each value 0-7 appears exactly once
39 * per row and once per column, ensuring uniform distribution and avoiding
40 * hot-shard clustering for structured hash inputs.
41 *
42 * Construction: entry (r, c) = (r + c) % 8 (cyclic Latin square).
43 */
44static const uint8_t shard_route_table[TTAK_POOL_SHARD_COUNT][TTAK_POOL_SHARD_COUNT] = {
45 /* col: 0 1 2 3 4 5 6 7 */
46 /* r=0 */ { 0, 1, 2, 3, 4, 5, 6, 7 },
47 /* r=1 */ { 1, 2, 3, 4, 5, 6, 7, 0 },
48 /* r=2 */ { 2, 3, 4, 5, 6, 7, 0, 1 },
49 /* r=3 */ { 3, 4, 5, 6, 7, 0, 1, 2 },
50 /* r=4 */ { 4, 5, 6, 7, 0, 1, 2, 3 },
51 /* r=5 */ { 5, 6, 7, 0, 1, 2, 3, 4 },
52 /* r=6 */ { 6, 7, 0, 1, 2, 3, 4, 5 },

56/**
57 * @brief Convert a task hash into (row, col) shard coordinates.
58 *
59 * Uses Fibonacci/multiplicative hashing with the 64-bit golden-ratio constant
60 * to project the hash into a bounded 2D coordinate space. The upper and lower
61 * halves of the mixed value are used independently to decorrelate the two axes.
62 *
63 * @param hash Task hash value.
64 * @param row Output: row coordinate in [0, TTAK_POOL_SHARD_COUNT).
65 * @param col Output: column coordinate in [0, TTAK_POOL_SHARD_COUNT).
66 */
67static inline void ttak_shard_hash_to_coords(uint64_t hash,
68 uint32_t *row,
69 uint32_t *col)
70{
71 /* 64-bit golden-ratio constant (⌊2^64 / φ⌋) for Fibonacci hashing */
72 const uint64_t GOLDEN = UINT64_C(0x9e3779b97f4a7c15);
73 uint64_t mixed = hash * GOLDEN;
74 *row = (uint32_t)(mixed >> 32) & (uint32_t)TTAK_POOL_SHARD_MASK;
75 *col = (uint32_t)(mixed & UINT32_MAX) & (uint32_t)TTAK_POOL_SHARD_MASK;
76 const uint32_t row_bits = *row & (uint32_t)TTAK_MOLS_SYMBOL_MASK;
77 const uint32_t col_bits = *col & (uint32_t)TTAK_MOLS_SYMBOL_MASK;
78 const uint16_t node_id =
79 (uint16_t)((row_bits << TTAK_MOLS_COORD_SHIFT) | col_bits);
80 const uint32_t combined =
81 (row_bits << TTAK_MOLS_COORD_SHIFT) | col_bits;
82 const uint32_t adjusted = ttak_apply_mols_control(node_id, combined);
83 *row = (adjusted >> TTAK_MOLS_COORD_SHIFT) & (uint32_t)TTAK_POOL_SHARD_MASK;
84 *col = adjusted & (uint32_t)TTAK_POOL_SHARD_MASK;
85}

87/**
88 * @brief Map a task hash to a shard index via the 2D routing table.
89 *
90 * This is the primary deterministic mapping function. Given any 64-bit
91 * hash the output is always in [0, TTAK_POOL_SHARD_COUNT) and is
92 * reproducible across runs.
93 *
94 * @param hash Task identifier hash.
95 * @return Shard index in [0, TTAK_POOL_SHARD_COUNT).
96 */
97static inline size_t ttak_shard_for_hash(uint64_t hash)
98{
99 uint32_t row, col;
100 ttak_shard_hash_to_coords(hash, &row, &col);
101 return (size_t)shard_route_table[row][col];
102}

104/**
105 * @brief Map a worker index to its preferred shard.
106 *
107 * Workers are spread across shards using modular arithmetic so that the
108 * initial affinity assignment is uniform regardless of thread count.
109 *
110 * @param worker_idx Zero-based worker index.
111 * @return Preferred shard index.
112 */
113static inline size_t ttak_shard_for_worker(size_t worker_idx)
114{
115 return worker_idx & (size_t)TTAK_POOL_SHARD_MASK;
116}
