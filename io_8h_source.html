42/**
43 * @brief Staging buffer that pins a user pointer into a detachable arena.

45 * Acquire with ttak_io_buffer_acquire() and release with

48typedef struct ttak_io_buffer {

54} ttak_io_buffer_t;

56/**
57 * @brief TTL-bounded guard wrapping a file descriptor.
58 *
59 * All I/O operations check the guard before touching the fd. Once
60 * @c expires_at is exceeded the guard is considered invalid and operations
61 * return @c TTAK_IO_ERR_EXPIRED_GUARD.
62 */
63typedef struct ttak_io_guard {
64 int fd; /**< Underlying file descriptor. */
65 ttak_owner_t *owner; /**< Owning arena/epoch context. */
66 uint64_t ttl_ns; /**< Configured time-to-live in nanoseconds. */
67 uint64_t expires_at; /**< Absolute expiry timestamp (ns). */
68 uint64_t last_used; /**< Timestamp of the last successful operation. */
69 bool closed; /**< True after ttak_io_guard_close(). */
70 char resource_tag[32]; /**< Diagnostic label for the guarded resource. */
71} ttak_io_guard_t;
